    name: studentName,
    age: age,
    courses: courses
};

// Function with types
function calculateAverage(scores: number[]): number {
    const sum = scores.reduce((total, score) => total + score, 0);
    return sum / scores.length;
}

console.log(`${student.name}'s average: ${calculateAverage(grades)}`);
